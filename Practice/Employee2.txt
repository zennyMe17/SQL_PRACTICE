
create table DEPARTMENT(
	Dname varchar(25),
  	Dnumber int primary key,
  	MgrSSn int,
  	MgrStartDAte varchar(25)
);

create table EMPLOYEE (
	Fname varchar(25),
  	Lname varchar(25),
  	SSN int primary key,
  	Addrs varchar(25),
  	sex varchar(25),
  	salary int,
  	SuperSSN int,
  	Dno int,
  	foreign key (Dno) references DEPARTMENT(Dnumber)
);

create table  DEPENDENT(
	Dname varchar(25),
  	ESSN int,
  	foreign key (ESSN) references EMPLOYEE(SSN),
  	foreign key (Dname) references DEPARTMENT(Dnumber)
);
